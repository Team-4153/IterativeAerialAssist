007package com.team4153.systems;
008
009import com.team4153.RobotMap;
010import com.team4153.Sensors;
011import edu.wpi.first.wpilibj.Joystick;
012import java.util.Vector;
013
014/**
015 *
016 * @author 4153student
017 */
018public class JoystickHandler implements Systems {
019    
020    /** Vector with all of the button handlers that control systems. */
021    Vector buttons=new Vector();
022    
023            
024    public JoystickHandler(Shooter shooter, Flippers flippers){
025        //Add all ButtonHandlers here
026        
027        buttons.addElement(new ButtonHandler(Sensors.getDriverJoystick(),RobotMap.JSBUTTON_GYRO_RESET,new GyroReset(),true));
028        buttons.addElement(new ButtonHandler(Sensors.getManipulatorJoystick(),RobotMap.JSBUTTON_FLIPPERS,flippers,true));
029        buttons.addElement(new ButtonHandler(Sensors.getManipulatorJoystick(),RobotMap.JSBUTTON_TRIGGER,shooter,true));
030    }
031    
032    public void execute(){
033        for(int i=0;i<buttons.size();i++){
034            ((ButtonHandler)(buttons.elementAt(i))).execute();
035        }
036    }
037    
038}
